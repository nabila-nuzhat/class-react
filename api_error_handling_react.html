// utils/apiError.js

// Custom error class to carry status + type
export class ApiError extends Error {
  constructor(message, type, status) {
    super(message);
    this.type   = type;   // 'network' | 'timeout' | 'abort' | 'http' | 'parse'
    this.status = status; // HTTP status code (if any)
  }
}

// Maps any thrown error → ApiError with a user-friendly message
export function classifyError(err, res) {
  if (res) {
    const messages = {
      400: "Invalid request. Please check your input.",
      401: "Session expired. Please sign in again.",
      403: "You don't have permission to do this.",
      404: "Resource not found.",
      408: "Request timed out. Please try again.",
      409: "Conflict — resource may already exist.",
      422: "Validation error. Check your data.",
      429: "Too many requests. Please wait a moment.",
      500: "Server error. We're working on it.",
      502: "Bad gateway. Server may be down.",
      503: "Service unavailable. Try again soon.",
      504: "Gateway timed out. Try again.",
    };
    const msg = messages[res.status] ?? `Unexpected error (${res.status})`;
    return new ApiError(msg, 'http', res.status);
  }

  if (err.name === 'AbortError') {
    return new ApiError(
      err._isTimeout
        ? "Request timed out. Check your connection."
        : "Request was cancelled.",
      err._isTimeout ? 'timeout' : 'abort'
    );
  }

  if (err.name === 'SyntaxError') {
    return new ApiError("Invalid response from server.", 'parse');
  }

  return new ApiError("Network error. Check your connection.", 'network');
}
// hooks/useFetch.js
import { useState, useEffect, useRef } from 'react';
import { classifyError } from '../utils/apiError';

export function useFetch(url, options = {}) {
  const [state, setState] = useState({
    data:    null,
    error:   null,   // ApiError | null
    loading: false,
  });
  const abortRef = useRef(null);

  useEffect(() => {
    const controller = new AbortController();
    abortRef.current = controller;

    // Optional: auto-abort after N ms
    const TIMEOUT_MS = options.timeout ?? 10_000;
    const timer = setTimeout(() => {
      controller.signal._isTimeout = true; // flag for classifier
      controller.abort();
    }, TIMEOUT_MS);

    async function fetchData() {
      setState(s => ({ ...s, loading: true, error: null }));
      let res = null;

      try {
        res = await fetch(url, { ...options, signal: controller.signal });

        if (!res.ok) throw new Error(); // trigger classifyError with res

        const data = await res.json(); // SyntaxError caught below
        setState({ data, error: null, loading: false });

      } catch (err) {
        if (controller.signal.aborted && !controller.signal._isTimeout) {
          return; // intentional cancel — don't update state
        }
        setState({
          data:    null,
          error:   classifyError(err, res?.ok === false ? res : null),
          loading: false,
        });
      } finally {
        clearTimeout(timer);
      }
    }

    fetchData();
    return () => controller.abort(); // cleanup on unmount
  }, [url]);

  return state; // { data, error, loading }
}

// components/ErrorMessage.jsx
const CONFIG = {
  network: { icon: '🌐', retry: true,  color: 'amber' },
  timeout: { icon: '⏱',  retry: true,  color: 'amber' },
  abort:   { icon: '✕',   retry: false, color: 'gray'  },
  parse:   { icon: '⚠',   retry: false, color: 'blue'  },
  http:    { icon: '⚡',  retry: true,  color: 'red'   },
};

// Retry-able status codes
const RETRYABLE = [408, 429, 500, 502, 503, 504];

export function ErrorMessage({ error, onRetry }) {
  if (!error) return null;

  const cfg = CONFIG[error.type] ?? CONFIG.http;
  const canRetry = cfg.retry
    && (typeof error.status === 'undefined' || RETRYABLE.includes(error.status));

  return (
    <div role="alert" className={`error-banner error-${cfg.color}`}>
      <span className="error-icon">{cfg.icon}</span>
      <div className="error-body">
        <p className="error-msg">{error.message}</p>
        {error.status && <p className="error-code">Code: {error.status}</p>}
      </div>
      {canRetry && onRetry && (
        <button onClick={onRetry} className="retry-btn">Retry</button>
      )}
    </div>
  );
}
